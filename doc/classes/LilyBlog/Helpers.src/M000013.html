    def class_for_tag(tag)
      case Post.tag(tag)
      when 0.0...0.02
        'tiny'
      when 0.02...0.04
        'small'
      when 0.04...0.06
        'medium'
      when 0.06...0.08
        'large'
      when 0.08..1.0
        'enormous'
      end
    end
