    def feed_urls
      urls = [[settings.posts_title, '/feed/']]

      if word = @tag || @search
        urls.push [settings.tag_title(word), "#{tag_url(word)}feed/"]
        urls.push [settings.search_title(word), "#{search_url(word)}feed/"]
      end

      return urls
    end
