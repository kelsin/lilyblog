    def initialize(file)
      @file = file

      File.open(file, 'r') do |file|
        data,@body = file.read.split(/\n(---)?\n/, 2)

        @meta = {}
        YAML::load(data).each do |key, val|
          @meta[key.to_sym] = val
        end
      end
    end
